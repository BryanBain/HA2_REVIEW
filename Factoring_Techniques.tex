\section{Answer Key}

\begin{enumerate}
	\item $(x+5)(x-3)$
    \item $(a-8)(a-7)$
    \item $(4x+3)(2x+1)$
    \item $(w+4)(w-3)$
	\item $(b-2)(5b+1)$
    \item $(2x+7)(6x-1)$
    \item $4(x-3)(x+2)$
    \item $(3t+1)(6t-5)$
    \item $(3a+7)(2a+3)$
    \item $(x-6)^2$
    \item $(3x-1)(3x+1)$
    \item $(2x+1)^2$
    \item $x(x-2)(x+1)$
    \item $2(3x-1)(x-5)$
    \item $(2x+5)(x+1)(x-8)$
    \item $(x+2)(2x+1)(x-1)$
    \item $(x+4)(4x-5)(x-2)$
    \item $3x(2x+7)(x-8)$
\end{enumerate}
