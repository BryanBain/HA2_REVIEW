\chapter{Compositions of Functions}

Given $f(x) = x^2 + 5$ and $g(x) = -3x-2$, find or evaluate each.

\begin{multicols}{4}
\begin{enumerate}
	\item $(f \circ g)(x)$
	\item $(g \circ f)(x)$
	\item $(f \circ f)(x)$
	\item $g(g(x))$
\end{enumerate}	\setcounter{Review}{\value{enumi}}
\end{multicols}
\begin{multicols}{4}
\begin{enumerate}		\setcounter{enumi}{\value{Review}}
	\item $(f \circ g)(1)$
	\item $(g \circ f)(-2)$
	\item $(f \circ f)(0)$
	\item $g(g(-8))$
\end{enumerate}	\setcounter{Review}{\value{enumi}}
\end{multicols}
