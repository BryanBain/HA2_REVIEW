\begin{multicols}{4}
\begin{enumerate}
	\item $(f \circ g)(x)$
	\item $(g \circ f)(x)$
	\item $(f \circ f)(x)$
	\item $g(g(x))$
\end{enumerate}	\setcounter{Review}{\value{enumi}}
\end{multicols}
\begin{multicols}{4}
\begin{enumerate}		\setcounter{enumi}{\value{Review}}

	\item $(f \circ f)(0)$
	\item $g(g(-8))$
\end{enumerate}	\setcounter{Review}{\value{enumi}}
