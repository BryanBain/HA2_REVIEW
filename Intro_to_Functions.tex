\chapter{Intro to Functions}

Evaluate each of the following given $f(x) = \frac{x}{5}+8$.
\begin{multicols}{3}
\begin{enumerate}
	\item $f(9)$
	\item $f(-1)$
	\item $f(8)$
\end{enumerate}	\setcounter{Review}{\value{enumi}}
\end{multicols}

\newpage

\section{Answer Key}

\begin{enumerate}
	\item $\frac{49}{5}$
    \item $\frac{39}{5}$
    \item $\frac{48}{5}$
\end{enumerate}
